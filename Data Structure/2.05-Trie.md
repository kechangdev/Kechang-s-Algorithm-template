## Trie

思想： 二维数组模拟一个树结构。



```c++
class Trie {
public:
    int idx, tr[N][M], cnt[N];
    void push(string& s) {
        int p = 0;
        for(auto& ch : s) {
            int u = ch - '0';
            if(!tr[p][u]) tr[p][u] = ++idx;
            p = tr[p][u];
        }
        ++cnt[p];
    }
    int count(string& s) {
        int p = 0;
        for(auto& ch : s) {
            int u = ch - '0';
            if(!tr[p][u]) return 0;
            p = tr[p][u];
        }
        return cnt[p];
    }
};
```
